// ════════════════════════════════════════════════════════════════════════════
// BinaryClockReader.Annotated.cs
// ════════════════════════════════════════════════════════════════════════════
//
// WHAT THIS FILE DOES:
//   Reads an MP4 video that has a rectangular barcode in its bottom-left corner.
//   The barcode contains two rows of equally-spaced black/white squares:
//     white square = bit 1,  dark square = bit 0.
//
//   Each row has 33 physical squares. The first square (index 0) is a sync
//   marker and is always discarded. The remaining 32 squares are data bits.
//
//   The library:
//     1. Crops the bottom-left 20% of every video frame
//     2. Finds the barcode by locating the interior HOLE in the saturation mask
//        (works regardless of border colour — red, brown, orange, etc.)
//     3. Cleans up the interior with Otsu's auto-threshold to eliminate gray
//        pixels caused by H.264 compression artifacts
//     4. Reads 32 data bits from each of the two rows, skipping the sync square
//     5. Optionally validates Row1 with a zero-parity XOR check
//     6. Writes a processed video and (optionally) a CSV of the bit values
//
// REQUIRED NUGET PACKAGES:
//   dotnet add package OpenCvSharp4
//   dotnet add package OpenCvSharp4.runtime.win              (Windows)
//   dotnet add package OpenCvSharp4.runtime.ubuntu.20.04-x64 (Linux)
//
// ════════════════════════════════════════════════════════════════════════════

using OpenCvSharp;   // OpenCV image-processing library for .NET
using System.Text;   // StringBuilder — used when building hex strings

public class BinaryClockReader
{
    // ══════════════════════════════════════════════════════════════════════════
    // CONFIGURATION PROPERTIES
    // These are the knobs you can adjust without touching any logic.
    // ══════════════════════════════════════════════════════════════════════════

    /// <summary>
    /// What fraction of the frame to keep, measured from the bottom-left.
    /// 0.20 = 20% of both width and height.
    /// Example: on a 1920×1080 video the crop region is 384×216 pixels.
    /// </summary>
    public double CropFraction { get; set; } = 0.20;

    /// <summary>
    /// Total number of physical squares per row INCLUDING the leading sync square.
    /// The sync square (index 0) is a fixed white/black alignment marker —
    /// it carries no data and is always skipped. The remaining 32 squares are
    /// the actual data bits returned per row.
    /// </summary>
    public int NumCols { get; set; } = 33;

    /// <summary>
    /// Size of the median blur kernel applied before Otsu thresholding.
    /// Median blur replaces each pixel with the median of its neighbourhood,
    /// which removes "salt-and-pepper" noise from H.264 compression while
    /// keeping the hard edges between black and white squares crisp.
    /// Must be an odd number. 1 = disabled, 3 = recommended.
    /// </summary>
    public int DenoiseSize { get; set; } = 3;

    /// <summary>
    /// Thickness in pixels used when redrawing the box border after processing.
    /// Without this redraw, BinariseInterior would paint the border black or white,
    /// erasing the visual marker from the output video.
    /// </summary>
    public int BorderThickness { get; set; } = 2;

    /// <summary>
    /// When voting on a cell's bit value, only the centre fraction of the cell
    /// is sampled. This avoids edge bleed from H.264 compression at cell boundaries.
    /// 0.5 = use the middle 50% of the cell area.
    /// </summary>
    public double SampleFraction { get; set; } = 0.5;

    /// <summary>
    /// Minimum fraction of sampled pixels that must be white for the cell to
    /// be read as bit 1. 0.5 = simple majority vote.
    /// Raise (e.g. to 0.7) if dark squares are being misread as 1.
    /// </summary>
    public double WhiteVoteThreshold { get; set; } = 0.5;

    /// <summary>
    /// Minimum area in pixels² for a contour hole to be considered as the
    /// barcode interior. Filters out tiny noise blobs and UI decoration.
    /// </summary>
    public double MinBoxArea { get; set; } = 500;

    /// <summary>
    /// Minimum width-to-height ratio of the interior hole.
    /// The barcode interior is always much wider than tall (33 cols × 2 rows).
    /// 10.0 comfortably below the expected ratio of 16.5.
    /// </summary>
    public double MinAspectRatio { get; set; } = 10.0;

    /// <summary>
    /// Maximum width-to-height ratio of the interior hole.
    /// 25.0 comfortably above 16.5, ruling out very thin horizontal structures.
    /// </summary>
    public double MaxAspectRatio { get; set; } = 25.0;

    /// <summary>
    /// The exact aspect ratio the interior hole is expected to have.
    /// 33 bit-squares wide ÷ 2 bit-squares tall = 16.5.
    /// Among all candidates that pass the min/max filter, the one whose ratio
    /// is CLOSEST to this value wins — a ratio of 16.5 is geometrically unique
    /// enough that nothing else in a typical video frame matches it.
    /// </summary>
    public double ExpectedAspectRatio { get; set; } = 16.5;

    /// <summary>
    /// Minimum HSV saturation (0–255) for a pixel to be treated as part of
    /// the coloured border. Works for any border colour:
    ///   Red   #FF0000  → S ≈ 255  ✓
    ///   Brown #502e1a  → S ≈ 171  ✓
    ///   Black square   → S ≈ 0    ✗ (excluded)
    ///   White square   → S ≈ 0    ✗ (excluded)
    ///   Gray background → S &lt; 30 ✗ (excluded)
    /// Lower (e.g. 25) if the border colour is very desaturated.
    /// Raise (e.g. 60) if background colours cause false hits.
    /// </summary>
    public int MinSaturation { get; set; } = 40;

    /// <summary>
    /// Minimum HSV brightness (0–255) for a pixel to be treated as border.
    /// Filters out near-black pixels that happen to have accidental saturation
    /// values due to compression noise.
    /// </summary>
    public int MinBorderValue { get; set; } = 20;

    /// <summary>
    /// Dilation kernel size applied to the saturation mask before contour detection.
    /// Must be odd. Default 3.
    /// Plain dilation (NOT morphological close) thickens the border ring to bridge
    /// small compression gaps WITHOUT filling the interior hole. The hole must
    /// stay dark so that CComp hierarchy can identify it as a child contour.
    /// Increase to 5 for heavily compressed video.
    /// </summary>
    public int DilateSize { get; set; } = 3;

    /// <summary>
    /// Pixels to expand the detected interior rect outward on each side to
    /// recover the full outer rectangle including the border stroke.
    /// Increase if the drawn border is thick or the detected rect clips it.
    /// </summary>
    public int BorderExpansion { get; set; } = 4;

    // ══════════════════════════════════════════════════════════════════════════
    // COLOUR CONSTANTS
    //
    // OpenCV stores colours in BGR order (Blue, Green, Red) — the opposite of
    // the familiar RGB. So (0, 0, 255) is pure red, not pure blue.
    //
    // The four-value BGRA version adds an Alpha channel:
    //   A = 255 → fully opaque
    //   A = 100 → semi-transparent (0x64 = 100 decimal)
    //
    // We need both BGR and BGRA because:
    //   MP4/AVI: no alpha channel — we use BGR (3 channels)
    //   MOV:     supports BGRA (4 channels) for transparency
    // ══════════════════════════════════════════════════════════════════════════

    private static readonly Scalar BlackBgr  = new(0,   0,   0);
    private static readonly Scalar BlackBgra = new(0,   0,   0,   255); // fully opaque black
    private static readonly Scalar RedBgr    = new(0,   0,   255);
    private static readonly Scalar RedBgra   = new(0,   0,   255, 100); // #FF000064
    private static readonly Scalar WhiteBgr  = new(255, 255, 255);
    private static readonly Scalar WhiteBgra = new(255, 255, 255, 255); // fully opaque white

    // ══════════════════════════════════════════════════════════════════════════
    // PUBLIC API
    // ══════════════════════════════════════════════════════════════════════════

    /// <summary>
    /// Crops the bottom-left region of every frame, binarises the barcode
    /// interior, and writes the processed video. No CSV is produced.
    /// </summary>
    public void CropVideo(string inputPath, string outputPath)
        => ProcessVideo(inputPath, outputPath, csvPath: null, boxColor: HighlightColor.Default);

    /// <summary>
    /// Same as CropVideo(input, output) but also writes a CSV with the decoded
    /// bit rows for every frame. Columns: frame, timestamp_ms, row0, row1,
    /// row0_hex, row1_hex.
    /// </summary>
    public void CropVideo(string inputPath, string outputPath, string csvPath)
        => ProcessVideo(inputPath, outputPath, csvPath, boxColor: HighlightColor.Default);

    /// <summary>
    /// Same as CropVideo(input, output) with a custom highlight colour drawn
    /// around the detected barcode box.
    /// </summary>
    public void CropVideo(string inputPath, string outputPath, HighlightColor boxColor)
        => ProcessVideo(inputPath, outputPath, csvPath: null, boxColor);

    /// <summary>
    /// Combines a custom highlight colour and CSV output.
    /// </summary>
    public void CropVideo(string inputPath, string outputPath, string csvPath, HighlightColor boxColor)
        => ProcessVideo(inputPath, outputPath, csvPath, boxColor);

    /// <summary>
    /// Detects the barcode rectangle by locating its interior HOLE in the
    /// HSV saturation mask rather than by tracing the coloured border.
    ///
    /// WHY HOLES INSTEAD OF BORDERS:
    ///   The border colour shifts with HDR tone-mapping and codec changes.
    ///   The interior is always the same regardless of border colour — it
    ///   contains only black/white squares (both S ≈ 0 in HSV).
    ///
    /// PIPELINE:
    ///   1. BGR → HSV; threshold on saturation → border becomes white ring,
    ///      interior + background become black.
    ///   2. Dilate to bridge compression gaps in the ring (keeps hole intact).
    ///   3. FindContours with CComp hierarchy: any contour whose Parent ≥ 0
    ///      is a HOLE (dark region inside a white ring). The barcode interior
    ///      is exactly that hole.
    ///   4. Score candidates by proximity to ExpectedAspectRatio (16.5).
    ///   5. Expand the winning hole rect outward by BorderExpansion pixels to
    ///      include the border stroke in the returned rectangle.
    /// </summary>
    public Rect? DetectBarcodeRect(Mat frame)
    {
        using var hsv     = new Mat();
        using var satMask = new Mat();
        using var dilated = new Mat();

        // Step 1: BGR → HSV.
        // HSV separates colour (Hue + Saturation) from brightness (Value).
        // S ≈ 0  = black, white, or gray pixel.
        // S >> 0 = any coloured pixel (red, brown, orange, etc.).
        Cv2.CvtColor(frame, hsv, ColorConversionCodes.BGR2HSV);

        // InRange: any hue (H 0–180), saturation ≥ MinSaturation, brightness ≥ MinBorderValue.
        // Result: border pixels → 255 (white), everything else → 0 (black).
        Cv2.InRange(hsv,
            lowerb: new Scalar(0,   MinSaturation, MinBorderValue),
            upperb: new Scalar(180, 255,           255),
            dst:    satMask);

        // Step 2: Dilation only (NOT morphological close).
        // Dilation grows each white pixel outward, bridging compression gaps
        // in the border ring. Because it only grows from existing white pixels,
        // it cannot fill the black interior — the hole remains intact.
        if (DilateSize > 1)
        {
            using var k = Cv2.GetStructuringElement(
                MorphShapes.Rect, new Size(DilateSize, DilateSize));
            Cv2.Dilate(satMask, dilated, k);
        }
        else
        {
            satMask.CopyTo(dilated);
        }

        // Step 3: CComp hierarchy contour detection.
        // CComp returns two levels:
        //   Level 0: outermost contours  → the white border rings
        //   Level 1: holes inside them   → the dark interior regions
        //
        // hierarchy[i] = [next, prev, firstChild, parent]
        //   parent >= 0 means this contour is nested inside another → it is a hole.
        //   parent <  0 means this contour is outermost → skip it.
        Cv2.FindContours(dilated, out Point[][] contours, out HierarchyIndex[] hierarchy,
            RetrievalModes.CComp, ContourApproximationModes.ApproxSimple);

        Rect?  best      = null;
        double bestScore = -1;

        for (int i = 0; i < contours.Length; i++)
        {
            // Only consider holes (child contours nested inside a border ring)
            if (hierarchy[i].Parent < 0) continue;

            double area = Cv2.ContourArea(contours[i]);
            if (area < MinBoxArea) continue;

            Rect   br    = Cv2.BoundingRect(contours[i]);
            double ratio = br.Width / (double)br.Height;

            if (ratio < MinAspectRatio || ratio > MaxAspectRatio) continue;

            // Similarity = min(ratio, expected) / max(ratio, expected)
            //   → 1.0 when ratio == ExpectedAspectRatio (perfect match)
            //   → approaches 0 as they diverge
            // Multiplied by area so a more complete hole beats a lucky fragment.
            double similarity = Math.Min(ratio, ExpectedAspectRatio)
                              / Math.Max(ratio, ExpectedAspectRatio);
            double score = similarity * area;

            if (score > bestScore)
            {
                bestScore = score;
                best      = br;
            }
        }

        // Step 4: Expand the interior rect outward to include the border stroke.
        if (best.HasValue)
        {
            var b = best.Value;
            best = ClampRect(
                new Rect(
                    b.X      - BorderExpansion,
                    b.Y      - BorderExpansion,
                    b.Width  + BorderExpansion * 2,
                    b.Height + BorderExpansion * 2),
                frame.Cols, frame.Rows);
        }

        return best;
    }

    /// <summary>
    /// Scans the video frame by frame and returns a <see cref="BarcodeFrameResult"/>
    /// for the first frame in which the barcode is successfully detected and
    /// both bit rows can be read. Parity is NOT checked.
    /// Returns null if no suitable frame is found.
    /// </summary>
    public BarcodeFrameResult? ExtractFirstFrame(string inputPath)
        => ExtractFirstFrameCore(inputPath, checkParity: false);

    /// <summary>
    /// Same as ExtractFirstFrame(inputPath) but when checkParity is true,
    /// any frame whose Row1 fails the zero-parity XOR check is skipped.
    ///
    /// PARITY CHECK RULE:
    ///   Split the 32 Row1 bits into eight 4-bit nibbles.
    ///   XOR all eight nibbles together. Result must be 0x0.
    ///   Any non-zero result means the bits are corrupted → skip the frame.
    /// </summary>
    public BarcodeFrameResult? ExtractFirstFrame(string inputPath, bool checkParity)
        => ExtractFirstFrameCore(inputPath, checkParity);

    /// <summary>
    /// Validates zero-parity of a 32-bit row.
    ///
    /// HOW IT WORKS:
    ///   XOR (exclusive-or) outputs 1 where inputs differ, 0 where they match.
    ///   XORing all eight nibbles together gives 0 if every bit position has
    ///   been set an even number of times — the "zero parity" condition.
    ///   Any single-bit corruption flips one nibble, making the result non-zero.
    ///
    ///   Example: 0xB ^ 0x4 ^ 0x... ^ last == 0x0 → valid ✓
    /// </summary>
    public static bool CheckParity(int[] bits)
    {
        if (bits.Length != 32)
            throw new ArgumentException($"Expected 32 bits for parity check, got {bits.Length}.");

        int accumulated = 0;

        // Process eight nibbles (4 bits each = 32 bits total)
        for (int i = 0; i < 32; i += 4)
        {
            // Reconstruct the 4-bit nibble value (0–15) from individual bits.
            // bits[i]   → most significant bit (value 8)
            // bits[i+3] → least significant bit (value 1)
            int nibble = (bits[i]     << 3)
                       | (bits[i + 1] << 2)
                       | (bits[i + 2] << 1)
                       |  bits[i + 3];

            // Accumulate with XOR. If all nibbles cancel out, result = 0.
            accumulated ^= nibble;
        }

        return accumulated == 0;
    }

    /// <summary>
    /// Reads the two rows of 32 data bits from within the detected box.
    ///
    /// Steps:
    ///   1. Inset the boxRect to exclude the border stroke pixels
    ///   2. Grayscale → median blur → Otsu threshold → clean binary image
    ///   3. Divide into a 2 × NumCols grid
    ///   4. For each cell, sample the centre SampleFraction and vote:
    ///      ≥ WhiteVoteThreshold white → bit 1, else → bit 0
    ///   5. Skip column 0 (sync square) and return 32 data bits per row
    ///
    /// Returns null if the interior region is too small to process.
    /// </summary>
    public BinaryRows? ReadBinaryRows(Mat frame, Rect boxRect)
    {
        // Inset so we don't read the border stroke pixels as data
        int inset    = BorderThickness + 1;
        var interior = ClampRect(new Rect(
            boxRect.X + inset,
            boxRect.Y + inset,
            Math.Max(1, boxRect.Width  - inset * 2),
            Math.Max(1, boxRect.Height - inset * 2)),
            frame.Cols, frame.Rows);

        if (interior.Width <= 4 || interior.Height <= 4)
            return null;

        // new Mat(frame, interior) is a sub-Mat — a zero-copy view into the
        // rectangular region. No pixel data is duplicated.
        using var interiorCropped  = new Mat(frame, interior);
        using var interiorGray     = new Mat();
        using var interiorDenoised = new Mat();
        using var interiorBinary   = new Mat();

        // Grayscale collapses RGB into one luminance value per pixel.
        // This eliminates any residual colour cast from compression.
        Cv2.CvtColor(interiorCropped, interiorGray, ColorConversionCodes.BGR2GRAY);

        // Median blur removes isolated bright/dark pixels from H.264 block
        // artifacts while keeping the hard edges between squares crisp.
        if (DenoiseSize > 1)
            Cv2.MedianBlur(interiorGray, interiorDenoised, DenoiseSize);
        else
            interiorGray.CopyTo(interiorDenoised);

        // Otsu's threshold automatically finds the best split between the dark
        // and light pixel clusters without a manually chosen cutoff value.
        // Result: every pixel is either exactly 0 or 255 — no grays.
        Cv2.Threshold(interiorDenoised, interiorBinary, 0, 255,
            ThresholdTypes.Binary | ThresholdTypes.Otsu);

        int cols = NumCols > 0 ? NumCols : DetectColumnCount(interiorBinary);
        if (cols <= 0) return null;

        var (row0, row1) = ExtractBits(interiorBinary, cols);
        return new BinaryRows(row0, row1);
    }

    // ══════════════════════════════════════════════════════════════════════════
    // PRIVATE IMPLEMENTATION
    // ══════════════════════════════════════════════════════════════════════════

    /// <summary>
    /// Shared core for both ExtractFirstFrame overloads.
    /// checkParity=false → return first readable frame.
    /// checkParity=true  → skip frames where Row1 XOR parity ≠ 0.
    /// </summary>
    private BarcodeFrameResult? ExtractFirstFrameCore(string inputPath, bool checkParity)
    {
        using var capture = new VideoCapture(inputPath);
        if (!capture.IsOpened())
            throw new Exception($"Cannot open video: {inputPath}");

        int    srcW = (int)capture.Get(VideoCaptureProperties.FrameWidth);
        int    srcH = (int)capture.Get(VideoCaptureProperties.FrameHeight);
        double fps  =      capture.Get(VideoCaptureProperties.Fps);

        int cropW = (int)(srcW * CropFraction);
        int cropH = (int)(srcH * CropFraction);
        int cropY = srcH - cropH;

        using var srcFrame = new Mat();
        using var cropped  = new Mat();

        int frameIdx = 0;
        while (capture.Read(srcFrame) && !srcFrame.Empty())
        {
            new Mat(srcFrame, new Rect(0, cropY, cropW, cropH)).CopyTo(cropped);

            Rect? box = DetectBarcodeRect(cropped);
            if (box.HasValue)
            {
                BinaryRows? rows = ReadBinaryRows(cropped, box.Value);
                if (rows != null)
                {
                    // If parity checking is requested, validate Row1 before
                    // accepting this frame. If it fails, continue scanning.
                    if (checkParity && !CheckParity(rows.Row1))
                    {
                        frameIdx++;
                        continue;
                    }

                    // TimeSpan.FromSeconds gives a human-readable timestamp
                    var timestamp = TimeSpan.FromSeconds(frameIdx / fps);
                    return new BarcodeFrameResult(frameIdx, timestamp, rows, box.Value);
                }
            }

            frameIdx++;
        }

        return null;
    }

    /// <summary>
    /// Core frame-by-frame processing loop shared by all CropVideo overloads.
    /// csvPath = null means no CSV output.
    /// </summary>
    private void ProcessVideo(string inputPath, string outputPath, string? csvPath, HighlightColor boxColor)
    {
        // wantAlpha: .mov supports 4-channel BGRA; .mp4/.avi silently drop alpha.
        bool wantAlpha = outputPath.EndsWith(".webm", StringComparison.OrdinalIgnoreCase)
                      || outputPath.EndsWith(".mov",  StringComparison.OrdinalIgnoreCase);

        using var capture = new VideoCapture(inputPath);
        if (!capture.IsOpened())
            throw new Exception($"Cannot open video: {inputPath}");

        // We cast to int/double because VideoCapture.Get() always returns double.
        int    srcW        = (int)capture.Get(VideoCaptureProperties.FrameWidth);
        int    srcH        = (int)capture.Get(VideoCaptureProperties.FrameHeight);
        double fps         =      capture.Get(VideoCaptureProperties.Fps);
        int    totalFrames = (int)capture.Get(VideoCaptureProperties.FrameCount);

        int cropW = (int)(srcW * CropFraction);
        int cropH = (int)(srcH * CropFraction);
        int cropX = 0;
        int cropY = srcH - cropH; // bottom-left: start from (frameH - cropH)

        Console.WriteLine($"Source      : {srcW}x{srcH}  @{fps:F2} fps  ({totalFrames} frames)");
        Console.WriteLine($"Crop region : x={cropX} y={cropY}  →  {cropW}x{cropH}");
        Console.WriteLine($"Alpha output: {wantAlpha}");
        Console.WriteLine($"Video out   : {outputPath}");
        if (csvPath != null) Console.WriteLine($"CSV out     : {csvPath}");
        Console.WriteLine();

        // FourCC: 4-character codec identifier.
        // "png " = lossless PNG inside .mov (preserves RGBA)
        // MP4V   = standard MPEG-4 for .mp4 (no alpha, lossy)
        FourCC fourcc = wantAlpha ? FourCC.FromString("png ") : FourCC.MP4V;

        using var writer = new VideoWriter(outputPath, fourcc, fps,
            new Size(cropW, cropH), isColor: true);
        if (!writer.IsOpened())
            throw new Exception("Cannot open VideoWriter. Check codec/path.");

        // Open CSV writer only when requested; null = skip CSV
        StreamWriter? csv = csvPath != null
            ? new StreamWriter(csvPath, append: false, Encoding.UTF8)
            : null;
        csv?.WriteLine("frame,timestamp_ms,row0,row1,row0_hex,row1_hex");

        // Allocate Mat objects outside the loop — reused every frame to avoid
        // repeated GC pressure from thousands of allocations per second.
        using var srcFrame  = new Mat();
        using var cropped   = new Mat();
        using var processed = new Mat();

        // Cache the last successfully detected box rect. If detection fails on
        // a blurry or transitional frame, the last good position is reused.
        Rect? boxRect = null;

        try
        {
            int frameIdx = 0;
            while (capture.Read(srcFrame) && !srcFrame.Empty())
            {
                double timestampMs = frameIdx * 1000.0 / fps;

                // Crop: new Mat(src, rect) is a zero-copy sub-Mat view.
                // CopyTo makes an independent copy we can safely modify.
                new Mat(srcFrame, new Rect(cropX, cropY, cropW, cropH)).CopyTo(cropped);

                // Detect and cache. A failed detection reuses the previous rect.
                Rect? detected = DetectBarcodeRect(cropped);
                if (detected.HasValue)
                    boxRect = detected.Value;

                // Start from a copy of the crop; only paint over the barcode area.
                if (wantAlpha)
                    Cv2.CvtColor(cropped, processed, ColorConversionCodes.BGR2BGRA);
                else
                    cropped.CopyTo(processed);

                if (boxRect.HasValue)
                {
                    // Paint the interior: dark → black, light → white
                    BinariseInterior(cropped, processed, boxRect.Value, wantAlpha);

                    if (csv != null)
                    {
                        BinaryRows? rows = ReadBinaryRows(cropped, boxRect.Value);
                        if (rows != null)
                        {
                            // string.Concat on int[] turns {1,0,1,1} → "1011"
                            csv.WriteLine(
                                $"{frameIdx},{timestampMs:F1}," +
                                $"{string.Concat(rows.Row0)},{string.Concat(rows.Row1)}," +
                                $"{rows.Row0Hex},{rows.Row1Hex}");
                        }
                    }

                    // Redraw the border. Without this, BinariseInterior would have
                    // painted the border pixels black or white.
                    Cv2.Rectangle(processed, boxRect.Value,
                        boxColor.ToScalar(wantAlpha), BorderThickness);
                }

                writer.Write(processed);
                frameIdx++;

                // \r overwrites the same console line — no new line per frame
                if (frameIdx % 30 == 0 || frameIdx == totalFrames)
                {
                    double pct = totalFrames > 0 ? frameIdx * 100.0 / totalFrames : 0;
                    Console.Write($"\r  Frame {frameIdx}/{totalFrames}  ({pct:F1}%)   ");
                }
            }

            Console.WriteLine($"\n\nDone! {frameIdx} frames written → {outputPath}");
            if (csvPath != null) Console.WriteLine($"Bits saved  → {csvPath}");
        }
        finally
        {
            // Always flush and close the CSV, even if an exception is thrown.
            // The ?. operator means "only call Dispose if csv is not null".
            csv?.Dispose();
        }
    }

    /// <summary>
    /// Binarises only the interior of the detected box within the output frame.
    /// Dark pixels → black, light pixels → white. Everything outside is unchanged.
    /// 'source' is read-only; 'target' is written to via a live sub-Mat ROI.
    /// </summary>
    private void BinariseInterior(Mat source, Mat target, Rect boxRect, bool wantAlpha)
    {
        int inset    = BorderThickness + 1;
        var interior = ClampRect(new Rect(
            boxRect.X + inset,
            boxRect.Y + inset,
            Math.Max(1, boxRect.Width  - inset * 2),
            Math.Max(1, boxRect.Height - inset * 2)),
            source.Cols, source.Rows);

        if (interior.Width <= 4 || interior.Height <= 4) return;

        using var interiorCropped  = new Mat(source, interior);
        using var interiorGray     = new Mat();
        using var interiorDenoised = new Mat();
        using var interiorBinary   = new Mat();
        using var interiorDark     = new Mat(); // 255 where pixel is dark
        using var interiorLight    = new Mat(); // 255 where pixel is light

        Cv2.CvtColor(interiorCropped, interiorGray, ColorConversionCodes.BGR2GRAY);
        if (DenoiseSize > 1)
            Cv2.MedianBlur(interiorGray, interiorDenoised, DenoiseSize);
        else
            interiorGray.CopyTo(interiorDenoised);

        Cv2.Threshold(interiorDenoised, interiorBinary, 0, 255,
            ThresholdTypes.Binary | ThresholdTypes.Otsu);

        // BinaryInv: pixels that were 0 (dark) become 255 in darkMask → paint black
        Cv2.Threshold(interiorBinary, interiorDark,  127, 255, ThresholdTypes.BinaryInv);
        // Binary:    pixels that were 255 (light) stay 255 in lightMask → paint white
        Cv2.Threshold(interiorBinary, interiorLight, 127, 255, ThresholdTypes.Binary);

        // new Mat(target, interior) is a LIVE VIEW into that region of target.
        // SetTo calls directly modify the pixels in target — no copy required.
        using var roi = new Mat(target, interior);
        roi.SetTo(wantAlpha ? BlackBgra : BlackBgr, interiorDark);
        roi.SetTo(wantAlpha ? WhiteBgra : WhiteBgr, interiorLight);
    }

    /// <summary>
    /// Divides the binary interior into a 2 × cols grid, samples the centre of
    /// each cell, and returns 32 data bits per row (column 0 sync is discarded).
    /// </summary>
    private (int[] row0, int[] row1) ExtractBits(Mat binary, int cols)
    {
        int w        = binary.Cols;
        int h        = binary.Rows;
        int rowH     = h / 2;    // top half = row0, bottom half = row1
        int colW     = w / cols; // ideal column width (last col takes remainder)
        int dataCols = cols - 1; // 33 total − 1 sync = 32 data bits
        var row0     = new int[dataCols];
        var row1     = new int[dataCols];

        for (int c = 0; c < cols; c++)
        {
            int cellX = c * colW;
            int cellW = (c == cols - 1) ? w - cellX : colW;

            for (int r = 0; r < 2; r++)
            {
                int cellY = r * rowH;
                int cellH = (r == 1) ? h - cellY : rowH;

                // Sample only the centre SampleFraction to avoid edge bleed
                int padX = (int)(cellW * (1 - SampleFraction) / 2);
                int padY = (int)(cellH * (1 - SampleFraction) / 2);

                var sampleRect = ClampRect(
                    new Rect(cellX + padX, cellY + padY,
                             Math.Max(1, cellW - padX * 2),
                             Math.Max(1, cellH - padY * 2)),
                    binary.Cols, binary.Rows);

                using var cell = new Mat(binary, sampleRect);

                // Val0 of Cv2.Mean on a single-channel binary image is the
                // average brightness: 0 = all black, 255 = all white.
                int bit = (Cv2.Mean(cell).Val0 / 255.0) >= WhiteVoteThreshold ? 1 : 0;

                // Column 0 is the sync square — skip it, don't store.
                if (c == 0) continue;

                // Store at index (c-1) so data starts at 0
                if (r == 0) row0[c - 1] = bit;
                else        row1[c - 1] = bit;
            }
        }

        return (row0, row1);
    }

    /// <summary>
    /// Auto-detects column count from the binary image by projecting vertically
    /// and counting bright runs. Used only when NumCols = 0.
    /// </summary>
    private static int DetectColumnCount(Mat binary)
    {
        using var colSum = new Mat();

        // Reduce collapses all rows into one by averaging: result is 1 × width.
        Cv2.Reduce(binary, colSum, ReduceDimension.Row, ReduceTypes.Avg, MatType.CV_32F);
        colSum.GetArray(out float[] vals);

        float mean = 0;
        for (int i = 0; i < vals.Length; i++) mean += vals[i];
        mean /= vals.Length;
        float threshold = mean * 0.5f; // column is "white" if above 50% of mean

        // Count transitions from dark to bright — each transition is one new square
        int  cols    = 0;
        bool inWhite = false;
        foreach (float v in vals)
        {
            if (!inWhite && v > threshold)      { cols++; inWhite = true; }
            else if (inWhite && v <= threshold) { inWhite = false; }
        }
        return cols;
    }

    /// <summary>
    /// Converts a bit array into an uppercase hex string. Processes 4 bits at a
    /// time (MSB first). 32 bits → 8 hex characters.
    /// Example: {1,0,1,1, 0,1,0,0} → "B4"
    /// </summary>
    private static string BitsToHex(int[] bits)
    {
        var sb = new StringBuilder(8);
        for (int i = 0; i < bits.Length; i += 4)
        {
            int nibble = 0;
            for (int b = 0; b < 4 && (i + b) < bits.Length; b++)
            {
                if (bits[i + b] == 1)
                    // 1 << (3-b): b=0 → value 8 (MSB), b=3 → value 1 (LSB)
                    nibble |= 1 << (3 - b);
            }
            sb.Append(nibble.ToString("X"));
        }
        return sb.ToString();
    }

    /// <summary>
    /// Clamps a Rect to fit within the given dimensions.
    /// Without this, any sub-Mat operation that goes outside the image bounds
    /// throws an OpenCV exception.
    /// </summary>
    private static Rect ClampRect(Rect r, int maxW, int maxH)
    {
        int x = Math.Max(0, r.X);
        int y = Math.Max(0, r.Y);
        int w = Math.Min(r.Width,  maxW - x);
        int h = Math.Min(r.Height, maxH - y);
        return new Rect(x, y, Math.Max(0, w), Math.Max(0, h));
    }
}

// ── HighlightColor ────────────────────────────────────────────────────────────

/// <summary>
/// An RGBA colour used to draw the detected barcode box border.
///
/// OpenCV uses BGR order internally, so we store colour in familiar RGBA
/// and convert to BGR(A) only when needed via ToScalar().
///
/// Create via:
///   HighlightColor.Default                   // semi-transparent red #FF000064
///   HighlightColor.FromHex("#FF000064")       // #RRGGBBAA
///   HighlightColor.FromRgba(255, 0, 0, 100)
/// </summary>
public readonly struct HighlightColor
{
    public byte R { get; }
    public byte G { get; }
    public byte B { get; }

    /// <summary>Alpha: 0 = fully transparent, 255 = fully opaque. Only preserved in .mov output.</summary>
    public byte A { get; }

    private HighlightColor(byte r, byte g, byte b, byte a) { R = r; G = g; B = b; A = a; }

    /// <summary>Semi-transparent red: #FF000064 (A = 0x64 = 100 ≈ 40% opacity).</summary>
    public static HighlightColor Default => new(255, 0, 0, 100);

    /// <summary>Creates a colour from RGBA byte components (0–255).</summary>
    public static HighlightColor FromRgba(byte r, byte g, byte b, byte a = 255)
        => new(r, g, b, a);

    /// <summary>
    /// Parses a hex colour string. Supported formats:
    ///   "#RGB"       → shorthand, each digit doubled (e.g. #F00 → #FF0000)
    ///   "#RRGGBB"    → alpha defaults to 255 (opaque)
    ///   "#RRGGBBAA"  → explicit alpha
    /// </summary>
    public static HighlightColor FromHex(string hex)
    {
        string h = hex.TrimStart('#');
        if (h.Length == 3) h = $"{h[0]}{h[0]}{h[1]}{h[1]}{h[2]}{h[2]}";
        if (h.Length != 6 && h.Length != 8)
            throw new ArgumentException($"Unrecognised hex colour: '{hex}'. Expected #RGB, #RRGGBB, or #RRGGBBAA.");

        // Convert.ToByte(str, 16) parses a 2-char hex pair → byte (e.g. "FF" → 255)
        byte r = Convert.ToByte(h[0..2], 16);
        byte g = Convert.ToByte(h[2..4], 16);
        byte b = Convert.ToByte(h[4..6], 16);
        byte a = h.Length == 8 ? Convert.ToByte(h[6..8], 16) : (byte)255;
        return new(r, g, b, a);
    }

    /// <summary>
    /// Returns an OpenCV Scalar in BGR or BGRA order.
    /// wantAlpha = true → include alpha channel (for .mov output).
    /// wantAlpha = false → 3-channel BGR only (for .mp4 output).
    /// </summary>
    public Scalar ToScalar(bool wantAlpha)
        => wantAlpha ? new Scalar(B, G, R, A) : new Scalar(B, G, R);

    public override string ToString() => $"RGBA({R},{G},{B},{A})";
}

// ── BinaryRows ────────────────────────────────────────────────────────────────

/// <summary>
/// Holds the decoded bit values for both rows of a barcode frame.
/// Row0 = top row, Row1 = bottom row.
/// Each array contains 32 data bits (sync square excluded).
/// 1 = white square, 0 = dark square.
/// </summary>
public class BinaryRows
{
    /// <summary>Top row as a 32-element bit array.</summary>
    public int[] Row0 { get; }

    /// <summary>Bottom row as a 32-element bit array.</summary>
    public int[] Row1 { get; }

    public BinaryRows(int[] row0, int[] row1) { Row0 = row0; Row1 = row1; }

    /// <summary>Row0 as a bit string e.g. "10110100..."</summary>
    public string Row0Bits => string.Concat(Row0);

    /// <summary>Row1 as a bit string.</summary>
    public string Row1Bits => string.Concat(Row1);

    /// <summary>Row0 as an 8-char uppercase hex string e.g. "B423F19C"</summary>
    public string Row0Hex => BitsToHex(Row0);

    /// <summary>Row1 as an 8-char uppercase hex string.</summary>
    public string Row1Hex => BitsToHex(Row1);

    private static string BitsToHex(int[] bits)
    {
        var sb = new StringBuilder(8);
        for (int i = 0; i < bits.Length; i += 4)
        {
            int nibble = 0;
            for (int b = 0; b < 4 && (i + b) < bits.Length; b++)
                if (bits[i + b] == 1) nibble |= 1 << (3 - b);
            sb.Append(nibble.ToString("X"));
        }
        return sb.ToString();
    }
}

// ── BarcodeFrameResult ────────────────────────────────────────────────────────

/// <summary>
/// Result returned by <see cref="BinaryClockReader.ExtractFirstFrame"/>.
/// Contains the frame index, wall-clock timestamp, decoded rows, and the
/// detected bounding rectangle within the cropped region.
/// </summary>
public class BarcodeFrameResult
{
    /// <summary>Zero-based index of this frame within the video.</summary>
    public int FrameIndex { get; }

    /// <summary>
    /// Wall-clock position calculated as FrameIndex / fps.
    /// Format with Timestamp.ToString(@"hh\:mm\:ss\.fff") for display.
    /// </summary>
    public TimeSpan Timestamp { get; }

    /// <summary>Decoded binary rows for this frame.</summary>
    public BinaryRows Rows { get; }

    /// <summary>
    /// Bounding rect of the barcode within the CROPPED region (not the full frame).
    /// To convert to full-frame Y: add (frameHeight × (1 − CropFraction)).
    /// </summary>
    public Rect BoxRect { get; }

    // Convenience pass-throughs into Rows
    public int[]  Row0     => Rows.Row0;
    public int[]  Row1     => Rows.Row1;
    public string Row0Bits => Rows.Row0Bits;
    public string Row1Bits => Rows.Row1Bits;
    public string Row0Hex  => Rows.Row0Hex;
    public string Row1Hex  => Rows.Row1Hex;

    public BarcodeFrameResult(int frameIndex, TimeSpan timestamp, BinaryRows rows, Rect boxRect)
    {
        FrameIndex = frameIndex;
        Timestamp  = timestamp;
        Rows       = rows;
        BoxRect    = boxRect;
    }

    public override string ToString() =>
        $"Frame {FrameIndex} @ {Timestamp:hh\\:mm\\:ss\\.fff} | " +
        $"Box {BoxRect.X},{BoxRect.Y} {BoxRect.Width}×{BoxRect.Height} | " +
        $"R0={Row0Hex} R1={Row1Hex}";
}
